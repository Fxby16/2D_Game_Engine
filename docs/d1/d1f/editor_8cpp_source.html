592 if(a.is_directory()&&!b.is_directory()){

596 return false;

600 );

612 m_CurrentPath=entry.path().string();

614 break;
615 }

617 if(ImGui::Selectable((ICON_FA_FILE+entry.path().filename().string()).c_str())){

620 if(file){

624 m_SelectedFileContent+=ch;
625 }
626 fclose(file);
627
628 isFileContentOpen=true;
629 }
630 }
631 }
632 }
633
634 if(isFileContentOpen && !m_SelectedFileContent.empty()){
635 ImGui::Begin("File Content",&isFileContentOpen);
636 ImGui::Text("%s",m_SelectedFileContent.c_str());
637 ImGui::End();
638 }else if(!m_SelectedFileContent.empty()){
